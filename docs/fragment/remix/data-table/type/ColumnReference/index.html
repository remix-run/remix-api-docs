ColumnReference

View Source

Summary

Typed reference to a table column.

Signature

type ColumnReference<tableName, columnName> =
  ColumnReferenceLike<`${tableName}.${columnName}`> & {
    [columnMetadataKey]: {
      columnName: columnName;
      qualifiedName: `${tableName}.${columnName}`;
      tableName: tableName;
    };
  };
